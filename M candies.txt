#include<bits/stdc++.h>
using namespace std;
#define ll long long
const ll M= 1e9+7;
 
ll n,k;  
vector<ll>v(101);
ll dp[101][100001];
 
ll solve(ll i, ll k)
{
    if(k==0) return 1;
 
    if(k<0) return 0;
 
    if(i==(n-1))
    {
        if(k<=v[i]&&k>=0) return dp[i][k]= 1;
        else return dp[i][k]=0;
    }
    if(dp[i][k]!=-1) return dp[i][k];
    
 
    return dp[i][k]=(M+solve(i+1,k)%M+solve(i,k-1)%M-solve(i+1,k-v[i]-1)%M)%M;
}
 
int main()
{
    cin>>n>>k;
    memset(dp,-1,sizeof(dp));
 
    for(ll i=0;i<n;i++)
    {
        cin>>v[i];
    }
    cout<<solve(0,k)%M<<endl;
}
