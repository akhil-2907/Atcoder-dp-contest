#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[100001];
vector<ll>v(101);
ll n,k;
ll solve(ll k)
{
    if(k==0) return 0;
 
    if(dp[k]!=-1) return dp[k];
 
    ll ans = 0;
 
    for(ll i=1;i<=n;i++)
    {
       if(k>=v[i])
       {
       	 if(solve(k-v[i])==0) ans =1;
       }
    }
    return dp[k]=ans;
}
 
int main()
{
    memset(dp,-1,sizeof(dp));
 
   cin>>n>>k;
 
    for(ll i=1;i<=n;i++)
    cin>>v[i];
    if(solve(k)==1) cout<<"First"<<endl;
    else cout<<"Second"<<endl;
   // for(ll i=1;i<=k;i++) cout<<i<<" "<<dp[i]<<endl;
}
