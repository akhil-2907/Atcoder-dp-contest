#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll arr[10001];
ll dp[10001][10001];
 
ll solve(int i,int j)
{
    if(i>j)  return 0;
 
    if(dp[i][j]!=-1) return dp[i][j];
 
    return dp[i][j]=max(arr[i]+min(solve(i+2,j),solve(i+1,j-1)),arr[j]+min(solve(i,j-2),solve(i+1,j-1)));
 
}
int main()
{
    int n; cin>>n;
    ll sum=0;
    
    for(int i=1;i<=n;i++) 
    {
    	cin>>arr[i];
    	sum+=arr[i];
    }
 
    memset(dp,-1,sizeof(dp));
 
 
    cout<<2*solve(1,n)-sum;
}
