#include<iostream>
using namespace std;
 
#define ll long long
 ll dp[105][100005];
 
 ll a[105];
 ll  b[105];
 
 ll solve(ll n,ll W)
 {
    for(ll i=0;i<=n;i++)
    {
        for(ll w=0;w<=W;w++)
        {
            if(i==0||w==0)
            {
                dp[i][w]=0;
            }
            else
            {
                ll inc =0,exc =0;
                if(a[i-1]<=w){
                inc=b[i-1]+dp[i-1][w-a[i-1]];
                }
                exc =dp[i-1][w];
        
                dp[i][w]=max(inc,exc);
            }
        }
    }
    return dp[n][W];
 }
 
 int main()
 {
     ll n,k;
     cin>>n>>k;
 
     for(ll i=0;i<n;i++)
     {
         cin>>a[i]>>b[i];
     }
     cout<<solve(n,k)<<endl;
 }
