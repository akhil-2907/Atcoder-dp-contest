#include<iostream>
using namespace std;
#define ll long long int
ll arr[1000001];
ll dp[1000001];
 
int solve(int n)
{
    dp[1]=0;
    dp[2]=abs(arr[2]-arr[1]);
 
    for(ll i=3;i<=n;i++)
    {
        ll opt1= dp[i-1]+abs(arr[i]-arr[i-1]);
        ll opt2= dp[i-2]+abs(arr[i]-arr[i-2]);
        dp[i]=min(opt1,opt2);
    }
    return dp[n];
 
}
int main()
{
    ll n;  cin>>n;
    
    for(int i=1;i<=n;i++)  cin>>arr[i];
 
    cout<<solve(n)<<endl;
}
